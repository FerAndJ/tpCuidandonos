int calcularTiempo([Parada], direccionFinal){
  int tiempoTotal = 0
  for(i=0. i< [Parada].lenght, i++){
     tiempoTotal = tiempoTotal + parada[i].tiempoRecorrido() + parada[i].getMinutosEspera()
  }
  tiempoTotal= tiempoTotal+direccionFinal.tiempoRecorrido()

  return tiempoTotal
}
